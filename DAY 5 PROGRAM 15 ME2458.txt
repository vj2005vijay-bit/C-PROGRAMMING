#include <stdio.h>
int main() {
    int arr[] = {2, 4, 6, 8, 10}; // Original array
    int size = sizeof(arr) / sizeof(arr[0]);
    int multiplier = 3;
    printf("Original array:\n");
    for (int i = 0; i < size; i++) {
        printf("%d ", arr[i]);
    }
    printf("\n\nArray after multiplication by %d:\n", multiplier);
    for (int i = 0; i < size; i++) {
        arr[i] *= multiplier;
        printf("%d ", arr[i]);
    }
    printf("\n");
    return 0;
}
