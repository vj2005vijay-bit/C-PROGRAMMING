#include<stdio.h>
int main() {
    int size, i;
    printf("Enter the number of elements in the arrays: ");
    scanf("%d", &size);
    int array1[size], array2[size], result[size];
    printf("Enter elements of the first array:\n");
    for(i = 0; i < size; i++) {
        scanf("%d", &array1[i]);
    }
    printf("Enter elements of the second array:\n");
    for(i = 0; i < size; i++) {
        scanf("%d", &array2[i]);
    }
    for(i = 0; i < size; i++) {
        result[i] = array1[i] - array2[i];
    }
    printf("Result of subtraction (array1 - array2):\n");
    for(i = 0; i < size; i++) {
        printf("%d ", result[i]);
    }
    return 0;
}
